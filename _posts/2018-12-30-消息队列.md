    - docker pull rabbitmq
- 启动
    - sudo docker run -d --hostname rabbit-alone --name rabbit-alone -p 15672:15672 -p 5672:5672  rabbitmq
- 进入对应容器
    - sudo docker exec -it rabbit-alone  /bin/bash
